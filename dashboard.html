// ====================================== // OM DIGITAL E-SEVA KENDRA // Dashboard JS // ====================================== // Check Login const user = JSON.parse(localStorage.getItem("user")); if (!user) { window.location.href = "login.html"; } // Show User document.getElementById("username").innerHTML = user.username + " (" + user.role + ")"; // Current Date document.getElementById("todayDate").innerHTML = new Date().toLocaleDateString("en-IN"); // Load Claims let claims = JSON.parse(localStorage.getItem("claims")) || []; // Today Entry document.getElementById("today").innerHTML = claims.length; // Total Amount let totalAmount = 0; claims.forEach(function(item){ totalAmount += Number(item.total || 0); }); document.getElementById("amount").innerHTML = "₹ " + totalAmount.toLocaleString(); // Recent Claim Table let table = document.getElementById("claimTable"); table.innerHTML = ""; if(claims.length==0){ table.innerHTML = ` No Claim Found `; }else{ claims.slice().reverse().forEach(function(item){ table.innerHTML += ` ${item.name} ${item.mh} ${item.taluka} ₹${item.total} ${item.date} `; }); } // Logout function logout(){ if(confirm("Are you sure you want to Logout?")){ localStorage.removeItem("user"); localStorage.removeItem("username"); localStorage.removeItem("role"); window.location.href="login.html"; } } // Auto Refresh setInterval(function(){ document.getElementById("todayDate").innerHTML = new Date().toLocaleDateString("en-IN"); },1000);
